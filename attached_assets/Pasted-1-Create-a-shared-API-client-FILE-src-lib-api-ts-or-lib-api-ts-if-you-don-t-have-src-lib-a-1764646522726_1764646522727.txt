1️⃣ Create a shared API client

FILE: src/lib/api.ts (or lib/api.ts if you don’t have src)

// lib/api.ts
import Constants from "expo-constants";

const API_BASE_URL =
  // You can set this in app.json -> extra.apiUrl for different envs
  (Constants.expoConfig?.extra as any)?.apiUrl ||
  "https://your-cloudflare-worker-url.com";

export type HttpMethod = "GET" | "POST" | "PUT" | "DELETE";

export interface ApiOptions {
  method?: HttpMethod;
  body?: any;
  token?: string | null;
  companyId?: string | null;
}

async function request<T = any>(
  path: string,
  { method = "GET", body, token, companyId }: ApiOptions = {}
): Promise<T> {
  const headers: Record<string, string> = {
    "Content-Type": "application/json",
  };

  if (token) headers["Authorization"] = `Bearer ${token}`;
  if (companyId) headers["x-company-id"] = companyId;

  const res = await fetch(`${API_BASE_URL}${path}`, {
    method,
    headers,
    body: body ? JSON.stringify(body) : undefined,
  });

  if (!res.ok) {
    const text = await res.text().catch(() => "");
    throw new Error(text || `API error: ${res.status}`);
  }

  if (res.status === 204) return {} as T;
  return (await res.json()) as T;
}

export const api = {
  get: <T>(path: string, opts?: Omit<ApiOptions, "method">) =>
    request<T>(path, { ...opts, method: "GET" }),
  post: <T>(path: string, body?: any, opts?: Omit<ApiOptions, "method" | "body">) =>
    request<T>(path, { ...opts, method: "POST", body }),
  put: <T>(path: string, body?: any, opts?: Omit<ApiOptions, "method" | "body">) =>
    request<T>(path, { ...opts, method: "PUT", body }),
  del: <T>(path: string, opts?: Omit<ApiOptions, "method">) =>
    request<T>(path, { ...opts, method: "DELETE" }),
};

// contexts/AuthContext.tsx
import React, {
  createContext,
  useContext,
  useState,
  useEffect,
  ReactNode,
} from "react";
import AsyncStorage from "@react-native-async-storage/async-storage";
import * as SecureStore from "expo-secure-store";
import { Platform } from "react-native";
import { api } from "@/lib/api";

export type UserType = "sales_rep" | "company" | "admin";

export interface User {
  id: string;
  email: string;
  name: string;
  userType: UserType;
  company?: string;
  phone?: string;
  createdAt: string;
  isAdmin?: boolean;
  companyId?: string; // <-- important
}

interface AuthContextType {
  user: User | null;
  token: string | null;
  companyId: string | null;
  isLoading: boolean;
  isAuthenticated: boolean;
  login: (email: string, password: string) => Promise<"2FA" | "OK" | "ERROR">;
  verify2FA: (email: string, code: string) => Promise<boolean>;
  signup: (data: SignupData) => Promise<boolean>;
  logout: () => Promise<void>;
}

interface SignupData {
  email: string;
  password: string;
  name: string;
  userType: UserType;
  company?: string;
  phone?: string;
}

// storage keys
const AUTH_USER_KEY = "auth_user";
const AUTH_TOKEN_KEY = "auth_token";

async function secureSet(key: string, value: string): Promise<void> {
  if (Platform.OS === "web") {
    await AsyncStorage.setItem(key, value);
  } else {
    await SecureStore.setItemAsync(key, value);
  }
}

async function secureGet(key: string): Promise<string | null> {
  if (Platform.OS === "web") {
    return AsyncStorage.getItem(key);
  } else {
    return SecureStore.getItemAsync(key);
  }
}

async function secureDelete(key: string): Promise<void> {
  if (Platform.OS === "web") {
    await AsyncStorage.removeItem(key);
  } else {
    await SecureStore.deleteItemAsync(key);
  }
}

const AuthContext = createContext<AuthContextType | undefined>(undefined);

export function AuthProvider({ children }: { children: ReactNode }) {
  const [user, setUser] = useState<User | null>(null);
  const [token, setToken] = useState<string | null>(null);
  const [companyId, setCompanyId] = useState<string | null>(null);
  const [isLoading, setIsLoading] = useState(true);

  useEffect(() => {
    loadSession();
  }, []);

  const loadSession = async () => {
    try {
      const [userData, tokenData] = await Promise.all([
        secureGet(AUTH_USER_KEY),
        secureGet(AUTH_TOKEN_KEY),
      ]);
      if (userData) {
        const parsed = JSON.parse(userData);
        setUser(parsed);
        setCompanyId(parsed.companyId || null);
      }
      if (tokenData) {
        setToken(tokenData);
      }
    } catch (error) {
      console.error("Error loading auth session:", error);
    } finally {
      setIsLoading(false);
    }
  };

  const login = async (
    email: string,
    password: string
  ): Promise<"2FA" | "OK" | "ERROR"> => {
    try {
      // step 1: /api/auth/login
      const res = await api.post<{
        success: boolean;
        requires2FA?: boolean;
      }>("/api/auth/login", { email, password });

      if (!res.success) return "ERROR";

      if (res.requires2FA) {
        return "2FA";
      }

      // in our backend design, login always requires 2FA,
      // but we'll keep this hook for future flexibility
      return "OK";
    } catch (e) {
      console.error("Login error:", e);
      return "ERROR";
    }
  };

  const verify2FA = async (email: string, code: string): Promise<boolean> => {
    try {
      const res = await api.post<{
        success: boolean;
        token: string;
        user: User;
      }>("/api/auth/verify-2fa", { email, code });

      if (!res.success) return false;

      setUser(res.user);
      setToken(res.token);
      setCompanyId(res.user.companyId || null);

      await Promise.all([
        secureSet(AUTH_USER_KEY, JSON.stringify(res.user)),
        secureSet(AUTH_TOKEN_KEY, res.token),
      ]);

      return true;
    } catch (e) {
      console.error("2FA verify error:", e);
      return false;
    }
  };

  const signup = async (data: SignupData): Promise<boolean> => {
    try {
      const res = await api.post<{ success: boolean }>(
        "/api/auth/register",
        data
      );
      return res.success;
    } catch (e) {
      console.error("Signup error:", e);
      return false;
    }
  };

  const logout = async () => {
    try {
      setUser(null);
      setToken(null);
      setCompanyId(null);
      await Promise.all([
        secureDelete(AUTH_USER_KEY),
        secureDelete(AUTH_TOKEN_KEY),
      ]);
    } catch (e) {
      console.error("Logout error:", e);
    }
  };

  return (
    <AuthContext.Provider
      value={{
        user,
        token,
        companyId,
        isLoading,
        isAuthenticated: !!user && !!token,
        login,
        verify2FA,
        signup,
        logout,
      }}
    >
      {children}
    </AuthContext.Provider>
  );
}

export function useAuth() {
  const ctx = useContext(AuthContext);
  if (!ctx) {
    throw new Error("useAuth must be used within an AuthProvider");
  }
  return ctx;
}


On the frontend login screen, you’ll now:

Call login(email, password) → if "2FA", show a 2FA screen

On 2FA screen, call verify2FA(email, code) → then navigate into app
