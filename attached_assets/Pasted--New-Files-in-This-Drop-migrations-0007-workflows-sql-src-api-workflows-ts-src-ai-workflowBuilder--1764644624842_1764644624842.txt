  created_by TEXT,

🔥 AI WORKFLOW BUILDER

src/ai/workflowBuilder.ts

Takes a user prompt and returns a full workflow blueprint.

import { runAI } from "../utils/ai";

export async function buildWorkflow(env: Env, input: string) {
  const prompt = `
You are WorkflowGPT, an AI that builds automated workflows for CRM, sales, call centers, and marketing.

Return JSON with:
- "name"
- "description"
- "steps": array of steps. Each step:
  - "type": trigger | condition | action | delay | branch
  - "config": JSON object describing event or action
`;

  const text = await runAI(env, prompt + "\nUser Prompt:\n" + input);

  try {
    return JSON.parse(text);
  } catch {
    return { name: "Untitled Workflow", description: "", steps: [] };
  }
}

🤖 AI WORKFLOW INTERPRETER

src/ai/workflowInterpreter.ts
Reads any event and decides which workflows should run.

import { runAI } from "../utils/ai";

export async function interpretEvent(env: Env, event: any, workflow: any) {
  const prompt = `
You are an AI automation interpreter.

Given this workflow step and this incoming event, decide if the step should activate.

Return JSON:
- "shouldRun": true/false
- "reason": explanation
`;

  const text = await runAI(env, prompt + `
Workflow Step: ${JSON.stringify(workflow)}
Event: ${JSON.stringify(event)}
  `);

  try {
    return JSON.parse(text);
  } catch {
    return { shouldRun: false, reason: "AI error" };
  }
}

⚡ WORKFLOW RUNNER

src/engine/runWorkflow.ts
Executes workflow step-by-step.

import { executeAction } from "./executeAction";
import { interpretEvent } from "../ai/workflowInterpreter";

export async function runWorkflow(env: Env, workflow: any, event: any) {
  for (const step of workflow.steps) {
    const decision = await interpretEvent(env, event, step);

    if (!decision.shouldRun) continue;

    if (step.type === "action") {
      await executeAction(env, step.config, event);
    }

    if (step.type === "delay") {
      const ms = step.config.ms || 0;
      await new Promise(r => setTimeout(r, ms));
    }

    // conditions and branches handled via interpreter logic
  }
}

export async function executeAction(env: Env, config: any, event: any) {
  switch (config.type) {
    case "send_sms":
      await fetch(env.SMS_API, {
        method: "POST",
        body: JSON.stringify({
          to: config.to || event.phone,
          message: config.message
        })
      });
      break;

    case "send_email":
      await fetch(env.EMAIL_API, {
        method: "POST",
        body: JSON.stringify({
          to: config.to || event.email,
          subject: config.subject,
          html: config.html
        })
      });
      break;

    case "log":
      console.log("[WORKFLOW LOG]", config.message, event);
      break;

    case "update_lead":
      await env.DB.prepare(
        "UPDATE leads SET status = ? WHERE id = ?"
      ).bind(config.status, event.leadId).run();
      break;

    case "assign_agent":
      await env.DB.prepare(
        "UPDATE leads SET assigned_to = ? WHERE id = ?"
      ).bind(config.agentId, event.leadId).run();
      break;

    case "simpletalk_call":
      await fetch(config.webhookUrl, {
        method: "POST",
        body: JSON.stringify({
          action: "start_call",
          number: config.number || event.phone
        })
      });
      break;

    case "create_task":
      await env.DB.prepare(
        "INSERT INTO tasks (id, deal_id, title, due_date) VALUES (?, ?, ?, ?)"
      )
        .bind(
          crypto.randomUUID(),
          config.dealId || event.dealId,
          config.title,
          config.due
        )
        .run();
      break;
  }
}

🔔 TRIGGER ROUTER

src/engine/triggerRouter.ts
Matches event → workflows.

export async function dispatchEvent(env: Env, event: any) {
  const workflows = await env.DB.prepare(
    "SELECT * FROM workflows WHERE active = 1"
  ).all();

  for (const wf of workflows.results) {
    const steps = await env.DB.prepare(
      "SELECT * FROM workflow_steps WHERE workflow_id = ? ORDER BY step_order ASC"
    ).bind(wf.id).all();

    wf.steps = steps.results.map(s => ({
      ...s,
      config: JSON.parse(s.config)
    }));

    await runWorkflow(env, wf, event);
  }
}

🌐 API: Create + Manage Workflows

src/api/workflows.ts

import router from "../router";
import { requireAuth } from "../middleware/auth";
import { json, unauthorized } from "../utils/response";
import { buildWorkflow } from "../ai/workflowBuilder";

router.post("/api/workflows/create", async (req, env) => {
  const user = await requireAuth(req, env);
  if (!user) return unauthorized();

  const { prompt } = await req.json();

  const wf = await buildWorkflow(env, prompt);

  const id = crypto.randomUUID();

  await env.DB.prepare(
    "INSERT INTO workflows (id, name, description, created_by) VALUES (?, ?, ?, ?)"
  ).bind(id, wf.name, wf.description, user.id).run();

  let order = 1;

  for (const step of wf.steps) {
    await env.DB.prepare(
      "INSERT INTO workflow_steps (id, workflow_id, step_order, type, config) VALUES (?, ?, ?, ?, ?)"
    )
      .bind(
        crypto.randomUUID(),
        id,
        order++,
        step.type,
        JSON.stringify(step.config)
      )
      .run();
  }

  return json({ success: true, workflowId: id });
});
