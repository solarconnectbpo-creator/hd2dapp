📄 FILE: src/ai/socialModel.ts

🔥 AI that generates hashtags, predicts virality, and content score.

import { runAI } from "../utils/ai";

export async function analyzePostContent(env: Env, content: string) {
  const prompt = `
You are an AI social media strategist for a sales-based community.

Analyze the following post content and return JSON with:
- "score": (0-100 viral potential)
- "hashtags": array of 5 suggested hashtags (lowercase, no # prefix)
- "sentiment": positive | neutral | negative
- "category": advice | win | question | story | motivation | other
- "notes": short guidance to improve reach

Post:
${content}
`;

  const text = await runAI(env, prompt);

  try {
    return JSON.parse(text);
  } catch {
    return {
      score: 50,
      hashtags: [],
      sentiment: "neutral",
      category: "other",
      notes: ""
    };
  }
}

📄 FILE: src/api/posts.ts

🔥 Full Twitter-style posting engine.

import router from "../router";
import { requireAuth } from "../middleware/auth";
import { json, badRequest, unauthorized } from "../utils/response";
import { analyzePostContent } from "../ai/socialModel";

// ----------------------------
// CREATE POST
// ----------------------------
router.post("/api/posts/create", async (req, env) => {
  const user = await requireAuth(req, env);
  if (!user) return unauthorized();

  const { content, mediaUrl } = await req.json();
  if (!content) return badRequest("Missing content");

  const id = crypto.randomUUID();

  // AI analysis
  const ai = await analyzePostContent(env, content);
  const hashtags = ai.hashtags.join(",");

  await env.DB.prepare(
    `INSERT INTO posts (id, user_id, content, media_url, hashtags)
     VALUES (?, ?, ?, ?, ?)`
  ).bind(id, user.id, content, mediaUrl || "", hashtags).run();

  return json({
    id,
    score: ai.score,
    hashtags: ai.hashtags,
    category: ai.category,
    notes: ai.notes
  });
});

// ----------------------------
// GET FEED (personalized)
// ----------------------------
router.get("/api/posts/feed", async (req, env) => {
  const user = await requireAuth(req, env);
  if (!user) return unauthorized();

  const posts = await env.DB.prepare(
    `
    SELECT p.*,
      (SELECT COUNT(*) FROM post_engagement WHERE post_id = p.id AND type = 'like') AS likes,
      (SELECT COUNT(*) FROM comments WHERE post_id = p.id) AS comments
    FROM posts p
    ORDER BY datetime(p.created_at) DESC
    LIMIT 50
    `
  ).all();

  return json(posts.results);
});

// ----------------------------
// LIKE POST
// ----------------------------
router.post("/api/posts/like", async (req, env) => {
  const user = await requireAuth(req, env);
  if (!user) return unauthorized();

  const { postId } = await req.json();

  await env.DB.prepare(
    `INSERT INTO post_engagement (id, post_id, user_id, type)
     VALUES (?, ?, ?, 'like')`
  ).bind(crypto.randomUUID(), postId, user.id).run();

  return json({ success: true });
});

// ----------------------------
// TRENDING POSTS
// ----------------------------
router.get("/api/posts/trending", async (req, env) => {
  const posts = await env.DB.prepare(
    `
    SELECT
      p.*,
      (
        (SELECT COUNT(*) FROM post_engagement WHERE post_id = p.id AND type = 'like') * 2 +
        (SELECT COUNT(*) FROM comments WHERE post_id = p.id) * 3
      ) AS score
    FROM posts p
    ORDER BY score DESC, p.created_at DESC
    LIMIT 25
    `
  ).all();

  return json(posts.results);
});

import router from "../router";
import { requireAuth } from "../middleware/auth";
import { json, unauthorized, badRequest } from "../utils/response";

router.post("/api/comments/create", async (req, env) => {
  const user = await requireAuth(req, env);
  if (!user) return unauthorized();

  const { postId, content } = await req.json();
  if (!content) return badRequest("Missing content");

  const id = crypto.randomUUID();

  await env.DB.prepare(
    "INSERT INTO comments (id, post_id, user_id, content) VALUES (?, ?, ?, ?)"
  ).bind(id, postId, user.id, content).run();

  return json({ success: true, id });
});

router.get("/api/comments/:postId", async (req, env) => {
  const user = await requireAuth(req, env);
  if (!user) return unauthorized();

  const postId = req.url.split("/").pop();

  const comments = await env.DB.prepare(
    `
    SELECT c.*, u.name AS user_name
    FROM comments c
    LEFT JOIN users u ON c.user_id = u.id
    WHERE post_id = ?
    ORDER BY datetime(c.created_at) ASC
    `
  ).bind(postId).all();

  return json(comments.results);
});
